name=opeNoise
qgisMinimumVersion=2.0
description=  Qgis Plugin to compute noise levels.
about= The plugin allows to compute the noise level generated by point sources or by road sources at fixed receiver points and buildings.
	Please consider the following semplifications:
	- the modelization is 2D;
	- only lateral diffractions is calculated (no vertical diffraction and reflections);
	- the terrain is flat.
version=1.0
